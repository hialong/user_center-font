public int fourSumCount(int[] nums1, int[] nums2, int[] nums3, int[] nums4) {
        HashMap<Integer, Integer> countAB = new HashMap<>();
        for(int u:nums1){
            for(int v:nums2){

                countAB.put(u+v,countAB.getOrDefault(u+v,0)+1);
            }
        }
        int ans = 0;
        for(int u:nums3){
            for(int v:nums4){
                if (countAB.containsKey(-u - v)) {
                    ans += countAB.get(-u - v);
                }
            }
        }
        return ans;
    }
